  55. Function getHTTPPage(url)  
  56.  'On Error Resume Next 
  57.  dim http  
  58.  set http=Server.createobject("Microsoft.XMLHTTP")  
  59.  Http.open "GET",url,false  
  60.  Http.send()  
  61.  if Http.readystate<>4 then 
  62.   exit function  
  63.  end if  
  64.  getHTTPPage=bytesToBSTR(Http.responseBody,"GB2312") 
  65.  set http=nothing 
  66.  If Err.number<>0 then  
  67.   getHTTPPage = "服务器获取文件内容出错"  
  68.   Err.Clear 
  69.  End If   
  70. End function 
  71. Function BytesToBstr(body,Cset) 
  72.  dim objstream 
  73.  set objstream = Server.CreateObject("adodb.stream") 
  74.  objstream.Type = 1 
  75.  objstream.Mode =3 
  76.  objstream.Open 
  77.  objstream.Write body 
  78.  objstream.Position = 0 
  79.  objstream.Type = 2 
  80.  objstream.Charset = Cset 
  81.  BytesToBstr = objstream.ReadText  
  82.  objstream.Close 
  83.  set objstream = nothing 
  84. End Function

  267. 文件对象读写操作
  268. <script language=vbs>
  269. on error resume next
  270. Set fso=CreateObject("Scripting.FileSystemObject") '使FSO组件可以被fso变量调用
  271. set link1=fso.GetFile("c:\windows\笔记本.txt").OpenAsTextStream(1/2/8,0/-1/-2)
  272. open1=link1.AtendOfline '判断文件指针是否以在行的末尾
  273. open2=link1.AtendOfstream '判断文件指针是否一再文件的末尾
  274. open3=link1.Column '返回当前字符所在文件的列号
  275. open4=link1.Line '返回当前字符所在文件的行号
  276. open5=link1.Read(10) '读取文件中指定数量的字符
  277. open6=link1.ReadAll '读取文件中所有的字符
  278. open7=link1.ReadLine '读取文件中一行中含有的字符
  279. open8=link1.Write "字符串" '将自定字符串或字符串变量写入文件
  280. open9=link1.WriteLine "字符" '将自定字符+一个换行符写入文件
  281. open10=link1.WriteBlankLines 10 '将指定数量的换行符写入文件
  282. open11=link1.Skip 10 '使文件指针跳过指定数量的字符
  283. open12=link1.SkipLine '使文件指针跳到下一行
  284. link1.Close '文件操作完毕,销毁文件对象指针
  285. Set fso=Nothing '释放fso变量与FSO组件的连接资源
  286. </script>
