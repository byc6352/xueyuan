为什么80%的码农都做不了架构师？>>>   


次缓冲区(或叫辅助缓冲区)尽管使用了波形文件自己的 TWaveFormatEx, 但最终播放的却只是 22050HZ 的 8 位立体声.
因为次缓冲区最终要混入主缓冲区才播放, 可主缓冲区的缺省格式是 22050HZ 的 8 位立体声(这利于在不同应用程序之间的平滑切换).

次缓冲区一旦建立, 其格式就无法修改了(无法使用缓冲区对象的 SetFormat() 方法); 好在主缓冲区可以重置格式.
也就是说, 播放 44100HZ、16 位的 Wave 时, 如果不通过主缓冲修改格式则无法原声播放.

要修改格式只能手动建立主缓冲区(我们无法使 DirectSound 自动建立的主缓冲区, 没有入口).

手动建立主缓冲区的注意事项:
1、SetCooperativeLevel(Handle, DSSCL_PRIORITY); 因为主缓冲应该是硬缓冲, 这会影响到其它应用程序.
2、为缓冲区指定 TDSBufferDesc 结构时须 TDSBufferDesc.dwFlags = DSBCAPS_PRIMARYBUFFER.
3、同时 TDSBufferDesc.dwBufferBytes = 0; 主缓冲使用的应该是硬缓冲, 其大小是固定的, 不能设置, 指定 0 即可
4、同时 TDSBufferDesc.lpwfxFormat = nil; 因为主缓冲区的格式已有默认, 重新设置必须使用 SetFormat() 方法.

另外, 主缓冲不支持 IDirectSoundBuffer8 接口(IDirectSoundBuffer8 比 IDirectSoundBuffer 多出一些功能);
在次缓冲中可以使用 IDirectSoundBuffer8，但不存在像 CreateSoundBuffer8 这样的函数, 可通过 IDirectSoundBuffer.QueryInterface() 方法方便获取.

测试程序: 
  1. unit Unit1;
  2.  
  3. interface
  4.  
  5. uses
  6.   Windows, Messages, SysUtils, Variants, Classes, Graphics, Controls, Forms,
  7.   Dialogs, StdCtrls;
  8.  
  9. type
  10.   TForm1 = class(TForm)
  11.     Button1: TButton;
  12.     Button2: TButton;
  13.     procedure FormCreate(Sender: TObject);
  14.     procedure FormDestroy(Sender: TObject);
  15.     procedure Button1Click(Sender: TObject);
  16.     procedure Button2Click(Sender: TObject);
  17.   end;
  18.  
  19. var
  20.   Form1: TForm1;
  21.  
  22. implementation
  23.  
  24. {$R *.dfm}
  25.  
  26. uses DirectSound, MMSystem;
  27.  
  28. var
  29.   myDSound: IDirectSound8;        //设备对象
  30.   bufPrimary: IDirectSoundBuffer; //主缓冲
  31.   buf: IDirectSoundBuffer;        //次缓冲
  32.   buf8: IDirectSoundBuffer8;      //次缓冲的 IDirectSoundBuffer8 接口
  33.  
  34. {初始化设备}
  35. procedure TForm1.FormCreate(Sender: TObject);
  36. begin
  37.   DirectSoundCreate8(nil, myDSound, nil);
  38.   {若手动建立主缓冲, 设备的优先级至少要指定为 DSSCL_PRIORITY}
  39.   myDSound.SetCooperativeLevel(Handle, DSSCL_PRIORITY);
  40. end;
  41.  
  42. {建立主缓冲, 并修改其格式}
  43. procedure TForm1.Button1Click(Sender: TObject);
  44. var
  45.   wavFormat,fmt2: TWaveFormatEx;
  46.   bufDesc: TDSBufferDesc;
  47. begin
  48.   ZeroMemory(@bufDesc, SizeOf(TDSBufferDesc));
  49.   bufDesc.dwSize := SizeOf(TDSBufferDesc);
  50.   bufDesc.dwFlags := DSBCAPS_PRIMARYBUFFER; //指明建立的是主缓冲
  51.   bufDesc.dwBufferBytes := 0; //主缓冲有固定的大小, 无需指定, 须是 0
  52.   bufDesc.lpwfxFormat := nil; //主缓冲有自己的格式, 修改它须通过 SetFormat() 方法
  53.  
  54.   myDSound.CreateSoundBuffer(bufDesc, bufPrimary, nil);
  55.  
  56.   {显示修改前主缓冲格式}
  57.   bufPrimary.GetFormat(@fmt2, SizeOf(TWaveFormatEx), nil);
  58.   ShowMessageFmt('主缓冲默认: %dHZ %d 位 %d 声道', [fmt2.nSamplesPerSec, fmt2.wBitsPerSample, fmt2.nChannels]);
  59.  
  60.   {修改主缓冲的格式}
  61.   ZeroMemory(@wavFormat, SizeOf(TWaveFormatEx));
  62.   with wavFormat do begin
  63.     wFormatTag := WAVE_FORMAT_PCM;
  64.     nChannels := 2;
  65.     nSamplesPerSec := 44100;
  66.     wBitsPerSample := 16;
  67.     nBlockAlign := wBitsPerSample * nChannels div 8;
  68.     nAvgBytesPerSec := nSamplesPerSec * nBlockAlign;
  69.   end;
  70.   bufPrimary.SetFormat(@wavFormat);
  71.  
  72.   {显示修改后主缓冲格式}
  73.   bufPrimary.GetFormat(@fmt2, SizeOf(TWaveFormatEx), nil);
  74.   ShowMessageFmt('主缓冲改后: %dHZ %d 位 %d 声道', [fmt2.nSamplesPerSec, fmt2.wBitsPerSample, fmt2.nChannels]);
  75. end;
  76.  
  77. {建立次缓冲, 同时获取个 IDirectSoundBuffer8 接口}
  78. procedure TForm1.Button2Click(Sender: TObject);
  79. var
  80.   wavFormat,fmt2: TWaveFormatEx;
  81.   bufDesc: TDSBufferDesc;
  82. begin
  83.   {为建立次缓冲准备格式}
  84.   ZeroMemory(@wavFormat, SizeOf(TWaveFormatEx));
  85.   with wavFormat do begin
  86.     wFormatTag := WAVE_FORMAT_PCM;
  87.     nChannels := 2;
  88.     nSamplesPerSec := 44100;
  89.     wBitsPerSample := 16;
  90.     nBlockAlign := wBitsPerSample * nChannels div 8;
  91.     nAvgBytesPerSec := nSamplesPerSec * nBlockAlign;
  92.   end;
  93.  
  94.   ZeroMemory(@bufDesc, SizeOf(TDSBufferDesc));
  95.   bufDesc.dwSize := SizeOf(TDSBufferDesc);
  96.   bufDesc.dwFlags := DSBCAPS_STATIC;
  97.   bufDesc.dwBufferBytes := 3 * wavFormat.nAvgBytesPerSec; //指定容纳 3 秒钟的波形数据
  98.   bufDesc.lpwfxFormat := @wavFormat;
  99.  
  100.   {建立 IDirectSoundBuffer, 并查看其格式}
  101.   myDSound.CreateSoundBuffer(bufDesc, buf, nil);
  102.   buf.GetFormat(@fmt2, SizeOf(TWaveFormatEx), nil);
  103.   ShowMessageFmt('buf: %dHZ %d 位 %d 声道', [fmt2.nSamplesPerSec, fmt2.wBitsPerSample, fmt2.nChannels]);
  104.  
  105.   {从 IDirectSoundBuffer 获取 IDirectSoundBuffer8, 并查看其格式}
  106.   buf.QueryInterface(IID_IDirectSoundBuffer8, buf8); //
  107.   ZeroMemory(@fmt2, SizeOf(TWaveFormatEx));
  108.   buf.GetFormat(@fmt2, SizeOf(TWaveFormatEx), nil);
  109.   ShowMessageFmt('buf8: %dHZ %d 位 %d 声道', [fmt2.nSamplesPerSec, fmt2.wBitsPerSample, fmt2.nChannels]);
  110. end;
  111.  
  112. procedure TForm1.FormDestroy(Sender: TObject);
  113. begin
  114.   bufPrimary := nil;
  115.   buf := nil;
  116.   buf8 := nil;
  117.   myDSound := nil;
  118. end;
  119.  
  120. end.
